0003-02-hive-离线数仓ODS

错误

hdfs dfs -rm -r /spark-history/application_1630214450313_0001_1
hdfs dfs -rm -r /tmp/logs/root/logs-tfile/application_1630214450313_0001/flink04_35686
hdfs dfs -rm -r /tmp/logs/root/logs-tfile/application_1630214450313_0001/flink05_46674

遇到这种情况的话，直接删除就好了

记住在导入的时候，如果资源不够的话就会一直卡在哪里，最好的办法就是关掉机器，重新加到内存然后在启动

ODS建表

create database warehouse;

1、用户行为日志

drop table if exists ods_log;
create external table ods_log
(
`line` string  comment '用户的一行数据'
)
partitioned by (`dt` string)
stored as 
inputformat 'com.hadoop.mapred.DeprecatedLzoTextInputFormat'
outputformat 'org.apache.hadoop.hive.ql.io.HiveIgnoreKeyTextOutputFormat'
location '/user/hive/warehouse/warehouse.db/ods/ods_log';

1.1 数据装载

load data inpath '/off_line_warehouse/data/log/off_line_topic_log/2021-09-02' into table ods_log partition(dt='2021-09-02');

1.2 脚本

#!/bin/bash

# 定义变量方便修改
APP=warehouse

# 如果是输入的日期按照取输入日期；如果没输入日期取当前时间的前一天
if [ -n "$1" ] ;then
   do_date=$1
else 
   do_date=`date -d "-1 day" +%F`
fi 

echo ================== 日志日期为 $do_date ==================
sql="
load data inpath '/off_line_warehouse/data/log/off_line_topic_log/$do_date' into table ${APP}.ods_log partition(dt='$do_date');
"

hive -e "$sql"

hadoop jar /opt/flink/platform/hadoop-3.1.3/share/hadoop/common/hadoop-lzo-0.4.20.jar com.hadoop.compression.lzo.DistributedLzoIndexer /user/hive/warehouse/warehouse.db/ods/ods_log/dt=$do_date

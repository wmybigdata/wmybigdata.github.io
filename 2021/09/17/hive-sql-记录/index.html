Hive SQL总结

1、内置运算

1.1 字符串运算符

1.1.1 length

-- 语法: length(string A)
select length('wmy-love-bigdata');

1.1.2 reverse

-- 语法: reverse(string A)
select reverse('wmy-love-bigdata');

1.1.3 concat

-- 语法: concat(string A, string B…)
select concat('wmy',',','bigdata');

1.1.4 concat_ws

-- 语法: concat_ws(string SEP, string A, string B…)
select concat_ws('===','wmy','bigdata');

1.1.5 substr,substring

-- 语法: substr(string A, int start, int len),substring(string A, int start, int len)
select substr('wmybigdatalove',0,10);
select substring('wmybigdatalove',0,10);

1.1.6 upper,ucase

-- 语法: upper(string A) ucase(string A)
select upper('wmy');
select ucase('wmy');

1.1.7 lower,lcase

-- 语法: lower(string A) lcase(string A)
select lower('WMY');
select lcase('WMY');

1.1.8 trim

-- 语法: trim(string A)
select trim(' wmy ')

1.1.9 ltrim

-- 语法: ltrim(string A)
select ltrim(' wmy');

1.1.10 rtrim

-- 语法: rtrim(string A)
select rtrim('wmy ');

1.1.11 regexp_replace

-- 语法：regexp_replace(string A, string B, string C)
select regexp_replace('wmy==bigdata','==','**');

1.1.12 regexp_extract

-- 语法: regexp_extract(string subject, string pattern, int index)
select regexp_extract('wmylovebigdata', 'wmy(.*?)(bigdata)', 1);

1.1.13 parse_url

-- 语法: parse_url(string urlString, string partToExtract [, string keyToExtract])
select parse_url('http://facebook.com/path1/p.php?k1=v1&k2=v2#Ref1', 'HOST');
select parse_url('http://facebook.com/path1/p.php?k1=v1&k2=v2#Ref1', 'QUERY', 'k1');

1.1.14 get_json_object

-- 语法: get_json_object(string json_string, string path)
select get_json_object('{"id":"1001","name":"wmy","age":22}','$.name');

1.1.15 space

-- 语法: space(int n)
select length(space(10));

1.1.16 repeat

-- 语法: repeat(string str, int n)
select repeat('wmy=',3);

1.1.17 ascii

/*
语法: ascii(string str)
返回值: int
说明：返回字符串str第一个字符的ascii码
*/
select ascii('wmy');

1.1.18 lpad

/*
语法: lpad(string str, int len, string pad)

返回值: string

说明：将str进行用pad进行左补足到len位
*/

select lpad('wmy',10,'|');

1.1.19 rpad

/*
语法: rpad(string str, int len, string pad)

返回值: string

说明：将str进行用pad进行右补足到len位
*/

select rpad('wmy',10,'|');

1.1.20 split

/*
语法: split(string str, string pat)

返回值: array

说明: 按照pat字符串分割str，会返回分割后的字符串数组
*/

select split('wmy-love-bigdata','-')[0],split('wmy-love-bigdata','-')[1],split('wmy-love-bigdata','-')[2];

1.1.21 find_in_set

/*
语法: find_in_set(string str, string strList)

返回值: int

说明: 返回str在strlist第一次出现的位置，strlist是用逗号分割的字符串。如果没有找该str字符，则返回0
*/

select find_in_set('wmy','wmy,love,bigdata');

1.1.22 instr

/*
语法：INSTR(string C1,string C2,int I,int J)
返回:int。substr在str中第一次出现的位置，若任何参数为null返回null，若substr不在str中返回0，Str中第一个字符的位置为1

说明：C1 被搜索的字符串

C2 希望搜索的字符串

I 搜索的开始位置,默认为1

J 出现的位置,默认为1
*/

select instr('love bigdata, wmy is a boy','wmy');

1.1.23 str_to_map

/*
使用两个分隔符将文本拆分为键值对
语法：str_to_map(text[, delimiter1, delimiter2])
返回：map

Delimiter1将文本分成K-V对，Delimiter2分割每个K-V对。对于delimiter1默认分隔符是','，对于delimiter2默认分隔符是'='
*/

select str_to_map('id:11@name:wmy@age:man','@',':');

1.1.24 unix_timestamp

/*
unix_timestamp() 返回当前时间戳。另外，current_timestamp() 也有同样作用。

unix_timestamp(string date) 返回 date 对应的时间戳，date 格式必须为 yyyy-MM-dd HH:mm:ss。

unix_timestamp(string date, string format) 返回 date 对应的时间戳，date 格式由 format 指定。
*/


select unix_timestamp();
select current_timestamp();

select unix_timestamp('2021-09-15 15:15:15','yyyy-MM-dd HH:mm:ss');

1.1.25 from_unixtime

/*
from_unixtime(int/bigint timestamp) 返回 timestamp 时间戳对应的日期，格式为 yyyy-MM-dd HH:mm:ss。

from_unixtime(int/bigint timestamp, string format) 返回 timestamp 时间戳对应的日期，格式由 format 指定。
*/

select from_unixtime(1631718915,'yyyy-MM-dd HH:mm:ss');

1.1.26 array_contains

/*
字符串数组，元素包含操作
语法：array_contains(array arr, element)
返回数组中是否包含element元素：true or false
*/

select if((select array_contains(split("aaa:bbb:ccc:ddd",":"), "ccc")),1,0);

2、日期函数

1、to_date

返回日期部分
to_date
select to_date("2021-01-01 00:00:00") ;

year | quarter | month | day | hour | minute | second | weekofyear | 

select quarter('2021-09-17');

从源（从 Hive 2.2.0 开始）检索字段，例如天数或小时数。来源必须是日期、时间戳、间隔或可以转换为日期或时间戳的字符串。支持的字段包括：天、星期几、小时、分钟、月、季度、秒、周和年。

select extract(month from "2016-10-20");

select extract(hour from "2016-10-20 05:06:07");

select extract(dayofweek from "2016-10-20 05:06:07");

-- 注意：一个星期是从周日开始的
select extract(month from interval '1-3' year to month);

select extract(minute from interval '3 12:20:30' day to second);

select assert_true (2>1);

5、get_json_object

select get_json_object(
'{"store":{"fruit":[{"weight":8,"type":"apple"}, {"weight":9,"type":"pear"}],  "bicycle":{"price":19.95,"color":"red"}}, "email":"amy@only_for_json_udf_test.net", "owner":"amy" }',
'$.store.fruit.weight');

只需要select get_json_object(fields,’$.field.field’)

get_json_object
支持 JSONPath 的有限版本：

$ : 根对象
. : 子运算符
[] : 数组的下标运算符
* : [] 的通配符
不支持的语法值得注意：

: 零长度字符串作为键
.. : 递归下降
@ ：当前对象/元素
() : 脚本表达式
?() : 过滤（脚本）表达式。
[,] : 联合运算符
[start:end.step] : 数组切片操作符

5、内置表生成函数（UDTF）

explode (array)

select 
id,
course
from 03_wmy_array_course
lateral view explode(choose_course) cc as course;

explode (map)

select explode(map('A',10,'B',20,'C',30));
select explode(map('A',10,'B',20,'C',30)) as (key,value);
select tf.* from (select 0) t lateral view explode(map('A',10,'B',20,'C',30)) tf;
select tf.* from (select 0) t lateral view explode(map('A',10,'B',20,'C',30)) tf as key,value;

posexplode (array)

select posexplode(array('A','B','C'));
select posexplode(array('A','B','C')) as (pos,val);
select tf.* from (select 0) t lateral view posexplode(array('A','B','C')) tf;
select tf.* from (select 0) t lateral view posexplode(array('A','B','C')) tf as pos,val;

inline (array of structs)

select inline(array(struct('A',10,date '2015-01-01'),struct('B',20,date '2016-02-02')));
